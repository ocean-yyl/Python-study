a = int('10011001', 2)  # 二进制转化为十进制
b = bin(a)  #十进制转化为二进制
a2 = int(b,2)   # 二进制转化为十进制
print("a = ",a)
print("b = ",b)
print("a2 = ",a2)


c = oct(a)  # 十进制整数转化为八进制
d = hex(a)  # 十进制整数转化为十六进制
print("c = ",c)
print("d = ",d)

a8 = int(c,8)   # 八进制转化为十进制
a16 = int(d,16) # 十六进制转化为十进制
print("a8 = ",a8)
print("a16 = ",a16)
